Date-range-picker examples

Set input selector:


                var dateRangePicker = new DateRangePicker({
                    elementSelector: '.my-selector',
                });

Set start month:


                var dateRangePicker = new DateRangePicker({
                    elementSelector: '.start-month-example',
                    month: 10
                });

Set start year:


                var dateRangePicker = new DateRangePicker({
                    elementSelector: '.start-year-example',
                    year: 2019
                });
